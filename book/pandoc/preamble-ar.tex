% Right-to-left Arabic typesetting, driven directly by polyglossia.
%
% We deliberately do NOT set pandoc's `lang`/`dir` metadata: pandoc would then
% load babel, and this TeX Live's babel is too old for the `provide=*` option
% pandoc emits. polyglossia gives clean XeLaTeX RTL without that.

% bidi (pulled in by polyglossia) insists hyperref is loaded *before* it.
% pandoc loads hyperref later (via \usepackage{bookmark}), so pre-load it here;
% its later bookmark/hypersetup then see hyperref already present.
\usepackage{hyperref}

\usepackage{polyglossia}
% Define the script fonts BEFORE \setmainlanguage, or polyglossia tries to use
% the current (math) font for Arabic and errors out.
\newfontfamily\arabicfont[Script=Arabic]{Amiri}
\newfontfamily\englishfont{STIX Two Text}
\setmainlanguage{arabic}
\setotherlanguage{english}
